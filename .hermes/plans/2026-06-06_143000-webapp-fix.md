```python
"""Helpers for reconstructing Pydantic models from DB dicts."""
from app.models import Activiteit2, Activiteit3

def reconstruct_activiteit_2(data: dict) -> Activiteit2:
    markeringen = [Markering(**m) if isinstance(m, dict) else m for m in data.get("markeringen", [])]
    leden = [LidMarkering(**l) if isinstance(l, dict) else l for l in data.get("leden", [])]
    return Activiteit2(markeringen=markeringen, leden=leden)

def reconstruct_activiteit_3(data: dict) -> Activiteit3:
    begrippen = [Begrip(**b) if isinstance(b, dict) else b for b in data.get("begrippen", [])]
    afleidingsregels = [Afleidingsregel(**a) if isinstance(a, dict) else a for a in data.get("afleidingsregels", [])]
    return Activiteit3(begrippen=begrippen, afleidingsregels=afleidingsregels)
```

## Fase 5: Performance

### Task 18: httpx AsyncClient pooling in llm.py

**Objective:** Eén module-level client hergebruikt over alle LLM calls.

**Files:**
- Modify: `webapp/app/engine/llm.py`

**Step 1:** Maak een module-level client met lifecycle:

```python
_client: httpx.AsyncClient | None = None

async def _get_client() -> httpx.AsyncClient:
    global _client
    if _client is None or _client.is_closed:
        _client = httpx.AsyncClient(timeout=120)
    return _client

async def shutdown_client():
    global _client
    if _client and not _client.is_closed:
        await _client.aclose()
```

**Step 2:** Gebruik `await _get_client()` in `_call_llm` ipv `async with httpx.AsyncClient(...)`.

**Step 3:** Register shutdown event in main.py:

```python
@app.on_event("shutdown")
async def shutdown():
    from app.engine.llm import shutdown_client
    await shutdown_client()
```

**Step 4:** Commit
